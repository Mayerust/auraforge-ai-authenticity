import requests

url = "https://your-app.onrender.com/analyze-audio"
headers = {"X-API-Key": "demo-key-123"}

with open("my_beat.mp3", "rb") as f:
    r = requests.post(url, headers=headers, files={"file": f})

result = r.json()
print(result["decision"])      # "BLOCK"
print(result["ai_probability"])  # 0.8832
const analyzeAudio = async (file) => {
  const form = new FormData();
  form.append("file", file);

  const res = await fetch("https://your-app.onrender.com/analyze-audio", {
    method: "POST",
    headers: { "X-API-Key": "demo-key-123" },
    body: form
  });

  const { ai_probability, decision } = await res.json();

  if (decision === "BLOCK") {
    rejectUpload("AI-generated audio detected");
  }
};
